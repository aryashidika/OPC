DO $$
BEGIN
  INSERT INTO upper_division_status (session_id, division, state)
  VALUES ({{this.params.sessionId}}, '{{this.params.division}}'::division_enum, 'REWORK'::division_state_enum)
  ON CONFLICT (session_id, division)
  DO UPDATE SET state = 'REWORK'::division_state_enum
  WHERE upper_division_status.state = 'SUBMITTED';

  IF '{{this.params.division}}' = 'COS' THEN
    UPDATE upper_division_status
    SET packages_ready_at = NULL
    WHERE session_id = {{this.params.sessionId}}
      AND division = 'COS'::division_enum
      AND packages_ready_at IS NOT NULL;
  END IF;
END $$;
