WITH cur AS (
  SELECT model_number FROM upper_article_session
  WHERE article_id = {{sel_article.selectedOptionValue}} AND is_latest = TRUE
  LIMIT 1
)
SELECT s.article_id, s.id AS session_id,
       COALESCE(ds.state::text, 'NOT_STARTED') AS division_state
FROM upper_article_session s
JOIN cur ON cur.model_number = s.model_number
LEFT JOIN upper_division_status ds
       ON ds.session_id = s.id AND ds.division = 'BLACKBOX'::division_enum
WHERE s.is_latest = TRUE AND s.status <> 'RELEASED'
  AND s.article_id <> {{sel_article.selectedOptionValue}}
ORDER BY s.article_id;
