SELECT
  p.id, p.part_id, p.part_name, p.material_desc,
  pp.cos_before_package_id, pp.cos_remark_before AS remark,
  cbp.package_no AS before_package_no, cbp.cos_type AS before_cos_type
FROM upper_part p
LEFT JOIN upper_part_process pp
  ON pp.session_id = p.session_id AND pp.part_id = p.part_id
LEFT JOIN cos_before_package cbp
  ON cbp.id = pp.cos_before_package_id
WHERE p.session_id = (SELECT id FROM v_current_session WHERE article_id = {{sel_article.selectedOptionValue}})
AND (p.delta_flag IS NULL OR p.delta_flag != 'REMOVED')
ORDER BY
  CAST(SPLIT_PART(p.part_id, '-', 1) AS INTEGER),
  CASE WHEN p.part_id LIKE '%-%' THEN SPLIT_PART(p.part_id, '-', 2)::int ELSE 0 END;
