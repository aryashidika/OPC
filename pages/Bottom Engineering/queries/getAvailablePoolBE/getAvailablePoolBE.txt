SELECT * FROM (
  SELECT
    'PART' AS kind, p.id AS ref, p.part_id AS code, p.part_id AS part_number,
    p.part_name AS part_name, p.part_name AS label,
    CASE WHEN wi_edit.id IS NOT NULL THEN true ELSE false END AS is_current_wip
  FROM upper_part p
  LEFT JOIN bottom_wip_input wi_edit
    ON wi_edit.input_kind = 'PART'
    AND wi_edit.input_ref = p.id
    AND wi_edit.wip_id = NULLIF({{appsmith.store.beEditWipId}}, 0)
  WHERE p.session_id = (
    SELECT id FROM v_current_session
    WHERE article_id = {{sel_article.selectedOptionValue}}
    LIMIT 1
  )
    AND p.is_bought_ready = FALSE
    AND p.is_sockliner = FALSE
    AND (p.delta_flag IS NULL OR p.delta_flag != 'REMOVED')
    AND (
      NOT EXISTS (
        SELECT 1 FROM bottom_wip_input wi
        WHERE wi.input_kind = 'PART'
          AND wi.input_ref = p.id
          AND wi.wip_id IS DISTINCT FROM NULLIF({{appsmith.store.beEditWipId}}, 0)
      )
      OR wi_edit.id IS NOT NULL
    )
  UNION ALL
  SELECT
    'WIP' AS kind, w.id AS ref, CAST(w.id AS TEXT) AS code,
    '#' || CAST(w.id AS TEXT) AS part_number, w.label AS part_name,
    '#' || CAST(w.id AS TEXT) || ' · ' || w.label || ' (' || w.wip_type || ')' AS label,
    CASE WHEN wi_edit.id IS NOT NULL THEN true ELSE false END AS is_current_wip
  FROM bottom_wip w
  LEFT JOIN bottom_wip_input wi_edit
    ON wi_edit.input_kind = 'WIP'
    AND wi_edit.input_ref = w.id
    AND wi_edit.wip_id = NULLIF({{appsmith.store.beEditWipId}}, 0)
  WHERE w.session_id = (
    SELECT id FROM v_current_session
    WHERE article_id = {{sel_article.selectedOptionValue}}
    LIMIT 1
  )
    AND w.id IS DISTINCT FROM NULLIF({{appsmith.store.beEditWipId}}, 0)
    AND (
      NOT EXISTS (
        SELECT 1 FROM bottom_wip_input wi
        WHERE wi.input_kind = 'WIP'
          AND wi.input_ref = w.id
          AND wi.wip_id IS DISTINCT FROM NULLIF({{appsmith.store.beEditWipId}}, 0)
      )
      OR wi_edit.id IS NOT NULL
    )
) AS pool
ORDER BY 
  is_current_wip DESC, kind,
  SPLIT_PART(code, '-', 1)::int,
  CASE WHEN code LIKE '%-%' THEN SPLIT_PART(code, '-', 2)::int ELSE 0 END;
