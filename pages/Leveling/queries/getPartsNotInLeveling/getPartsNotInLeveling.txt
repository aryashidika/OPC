SELECT p.id AS part_db_id, p.part_id, p.part_name
FROM upper_part p
WHERE p.session_id = (SELECT id FROM v_current_session WHERE article_id = {{sel_article.selectedOptionValue}})
AND (p.delta_flag IS NULL OR p.delta_flag != 'REMOVED')
AND NOT EXISTS (
  SELECT 1 FROM leveling_node_prb ln
  WHERE ln.session_id = (SELECT id FROM v_current_session WHERE article_id = {{sel_article.selectedOptionValue}})
  AND ln.part_id = p.part_id
)
ORDER BY SPLIT_PART(p.part_id, '-', 1)::int,
  CASE WHEN p.part_id LIKE '%-%' THEN SPLIT_PART(p.part_id, '-', 2)::int ELSE 0 END;
