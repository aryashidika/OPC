WITH src AS (
  SELECT pp.part_id
  FROM upper_part_process pp
  WHERE pp.session_id = {{this.params.oldSessionId}}
    AND (pp.laminating_machine IS NOT NULL OR pp.cutting_type IS NOT NULL OR pp.cos_type IS NOT NULL)
),
tgt_parts AS (
  SELECT part_id FROM upper_part WHERE session_id = {{this.params.newSessionId}}
),
tgt_existing AS (
  SELECT pp.part_id
  FROM upper_part_process pp
  WHERE pp.session_id = {{this.params.newSessionId}}
    AND (pp.laminating_machine IS NOT NULL OR pp.cutting_type IS NOT NULL OR pp.cos_type IS NOT NULL)
)
SELECT
  (SELECT COUNT(*) FROM src s JOIN tgt_parts t ON t.part_id = s.part_id JOIN tgt_existing e ON e.part_id = s.part_id) AS will_overwrite,
  (SELECT COUNT(*) FROM src s JOIN tgt_parts t ON t.part_id = s.part_id WHERE s.part_id NOT IN (SELECT part_id FROM tgt_existing)) AS will_fill_new,
  (SELECT COUNT(*) FROM src s WHERE s.part_id NOT IN (SELECT part_id FROM tgt_parts)) AS not_relevant;
