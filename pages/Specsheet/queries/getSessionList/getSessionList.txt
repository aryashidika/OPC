SELECT 
  s.id,
  s.article_id,
  s.model_number,
  m.model_name,
  s.revision_no,
  s.status,
  s.is_latest,
  TO_CHAR(s.created_at AT TIME ZONE 'Asia/Jakarta', 'DD Mon YYYY HH24:MI') AS created_at
FROM upper_article_session s
LEFT JOIN upper_model m
       ON m.model_number = s.model_number
      AND m.season       = s.season
ORDER BY s.created_at DESC;
